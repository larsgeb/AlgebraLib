113  try {

115  if (entry != 0)
116  P[row.first] = entry;
117  } catch (const std::exception &e) {
118  /*
119  * Nothing fancy, we're trying to acces a const Matrix. Right.. So what happens if we don't know which
120  * getColumn entries are exactly non-zero and we have to iterate? Well, normally, map[] would create a new map
121  * entry, but that is not allowed for a const! So we try to acces it with map::at(), which return an
122  * exception when an element is non-existent in map. Here we catch it and we... don't do anything with it!
