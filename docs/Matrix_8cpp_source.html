183  Matrix Matrix::CholeskyDecompose() const {
184  if (rows() != columns()) {
185  throw std::length_error("Matrix trace: matrix is not square.");
186  }
187 
188  Matrix LowerCholesky(rows(), columns());
189 
190  LowerCholesky[0][0] = sqrt((*this)[0][0]);
191 
192  for (int row = 1; row < rows(); ++row) {
193  for (int column = 0; column <= row; ++column) {
194 
195  double sum = 0;
196  for (int k = 0; k < column; k++) {
197  sum += LowerCholesky[row][k] * LowerCholesky[column][k];
198  }
199 
200  if (column == row) {
201  LowerCholesky[row][column] = sqrt((*this)[row][column] - sum);
202  } else {
203  LowerCholesky[row][column] = (1 / LowerCholesky[column][column]) * ((*this)[row][column] - sum);
204  }
205  }
206  }
